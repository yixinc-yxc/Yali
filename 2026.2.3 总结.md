## B

有点思维。

就是首先我们发现大于 $h$ 或 $l$ 的一定没有用。

所以我们统计小于等于 $h$ 或 $l$ 的数量，记为 $cnth$ and $antl$。

最后将 $\min(cnth, \dfrac{cntl}2)$ 输出就行了。

```cpp
const int N = 110;
int t, ans, n, h, l, a[N];

    ans = -inf;
    std::cin >> n >> h >> l;
```

## C

依旧爽到起飞。

存一个原数组副本 $b$ 。

先将 $b$ 数组从大到小排序，然后再从找到第一个 $a_i != b_i$ 的位置，记为 $pos1$，然后从这个位置开始，找到最后一个最大值的位置，记为 $pos2$ 最后翻转原数组的 $pos1$ 至 $pos2$ 就好了。

```cpp

using ull = unsigned long long;
using i128 = __int128;
using db = double;
#define rep(i, a, b) for (int i = a; i <= b; i++)
#define frep(i, b, a) for (int i = b; i >= a; i--)
#define all(x) (x).begin(), (x).end()
#define pb push_back
#define pf push_front
#define mp make_pair
#define s second
#define f first
const int inf = 1e9;
const ll INF = 1LL << 60LL;
const int mod = 998244353;
const db eps = 1e-9;
std::mt19937_64 rng(std::chrono::system_clock::now().time_since_epoch().count());

const int N = 2e5 + 10;
int t, n, pos, a[N], b[N];

void solve() {
    std::cin >> n;
    pos = -1;
    rep(i, 1, n) std::cin >> a[i], b[i] = a[i];
    std::sort(b + 1, b + n + 1, std::greater<int>());
    rep(i, 1, n) if (a[i] != b[i]) {
        pos = i;
        break;
    }
    if (pos == -1) {
        rep(i, 1, n) std::cout << a[i] << " \n"[i == n];
        return;
    }
    int pos1 = 0, mx = 0; 
    rep(i, pos, n) if (a[i] >= mx) mx = a[i], pos1 = i;
    std::reverse(a + pos, a + pos1 + 1);
    rep(i, 1, n) std::cout << a[i] << ' ';
    std::cout << '\n';
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr), std::cout.tie(nullptr);
    std::cin >> t;
    while (t--) solve();
}
```

## D

神人题目。

~~不会有小朋友写DS吧~~

第一眼看：哇，尽然是传说中的单点修改，区间查询，线段树，走你。

第二眼，这不是一眼后缀最大值吗。

先求出 $a$ 的后缀最大值，再求出 $b$ 的后缀最大值。

之后对于 $a$ , $b$ 的后缀最大值的最大值做前缀和就行了。

```cpp
#include <bits/stdc++.h>

using ll = long long;
using ull = unsigned long long;
using i128 = __int128;
using db = double;
#define rep(i, a, b) for (int i = a; i <= b; i++)
#define frep(i, b, a) for (int i = b; i >= a; i--)
#define all(x) (x).begin(), (x).end()
#define pb push_back
#define pf push_front
#define mp make_pair
#define s second
#define f first
const int inf = 1e9;
const ll INF = 1LL << 60LL;
const int mod = 998244353;
const db eps = 1e-9;
std::mt19937_64 rng(std::chrono::system_clock::now().time_since_epoch().count());

const int N = 2e5 + 10;
int t, n, q, b[N], a[N], rmx[N], rmx_[N], s[N];

void solve() {
    std::cin >> n >> q;
    std::fill(rmx + 1, rmx + n + 2, 0);
    std::fill(rmx_ + 1, rmx_ + n + 2, 0);
    rep(i, 1, n) std::cin >> a[i];
    rep(i, 1, n) std::cin >> b[i];
    frep(i, n, 1) rmx[i] = std::max(rmx[i + 1], a[i]);
    frep(i, n, 1) rmx_[i] = std::max({rmx_[i + 1], b[i]});
    rep(i, 1, n) s[i] = s[i - 1] + std::max(rmx[i], rmx_[i]);
    while (q--) {
        int l, r;
        std::cin >> l >> r;
        std::cout << s[r] - s[l - 1] << ' ';
    }
    std::cout << '\n';
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr), std::cout.tie(nullptr);
    std::cin >> t;
    while (t--) solve();
}
```

## E

神仙题目。

再也不起飞了。

一开始看到题目：这不是裸裸的二分吗？但是发现好像只能二分难度，而二分难度总分好像又要二分。
不想二分了。

先注意到最终难度总为 $a$ 中的数，但是好像没用。

然后注意到是一关一关往后打，先对 $b$ 求前缀和。

然后就牢底坐穿。

突然发现对 $a$ 排序后知道关卡编号，难度是可以算的。

设当前到了第 $i$ 关，打通这一关需要砍 $x$ 刀，根据前面的结论，设我们所选的难度为 $a_{pos}$ 则 $pos$ 一定满足等于总刀的数量减去 $x$ ，即为 $n - x + 1$。

否则就会刀不够用。

```cpp
#include <bits/stdc++.h>

using ll = long long;
using ull = unsigned long long;
using i128 = __int128;
using db = double;
#define rep(i, a, b) for (int i = a; i <= b; i++)
#define frep(i, b, a) for (int i = b; i >= a; i--)
#define all(x) (x).begin(), (x).end()
#define pb push_back
#define pf push_front
#define mp make_pair
#define s second
#define f first
const int inf = 1e9;
const ll INF = 1LL << 60LL;
const int mod = 998244353;
const db eps = 1e-9;
std::mt19937_64 rng(std::chrono::system_clock::now().time_since_epoch().count());

const int N = 2e5 + 10;
int t, n, a[N], b[N];
ll ans, s[N];

void solve() {
    std::cin >> n;
    ans = 0;
    rep(i, 1, n) std::cin >> a[i];
    rep(i, 1, n) std::cin >> b[i];
    rep(i, 1, n) s[i] = s[i - 1] + b[i];
    std::sort(a + 1, a + n + 1);
    rep(i, 1, n) if (s[i] <= n) ans = std::max(ans, 1ll * a[n - s[i] + 1] * i);
    std::cout << ans << '\n';
}

// 发现只能从 A 中选难度，不然无意义

    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr), std::cout.tie(nullptr);
    std::cin >> t;
    while (t--) solve();
}
```

## F

又爽飞了。

套路题。

就是将 $a$ 离散化，对于每个 $i(1 \leq i \leq n)$ ，枚举数组 $a$ 中的 $v$ ，更新 $dis_{i * v}$

```cpp
#include <bits/stdc++.h>

using ll = long long;
using ull = unsigned long long;
using i128 = __int128;
using db = double;
#define rep(i, a, b) for (int i = a; i <= b; i++)
#define frep(i, b, a) for (int i = b; i >= a; i--)
#define all(x) (x).begin(), (x).end()
#define pb push_back
#define pf push_front
#define mp make_pair
#define s second
#define f first
const int inf = 1e9;

const ll INF = 1LL << 60LL;
const int mod = 998244353;
const db eps = 1e-9;
std::mt19937_64 rng(std::chrono::system_clock::now().time_since_epoch().count());

const int N = 3e5 + 10;
int t, n, a[N], dis[N];
std::vector<int> b;

void solve() {
    std::cin >> n;
    b.clear();
    std::fill(dis + 1, dis + n + 1, inf);
    rep(i, 1, n) std::cin >> a[i], dis[a[i]] = 1, b.pb(a[i]);
    std::sort(all(b));
    b.erase(std::unique(all(b)), b.end());
    rep(i, 1, n) if (dis[i]) for (auto &v : b) if (1ll * v * i <= n) dis[v * i] = std::min(dis[v * i], dis[i] + 1);
    else break;
    rep(i, 1, n) std::cout << (dis[i] == inf ? -1 : dis[i]) << " \n"[i == n];
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr), std::cout.tie(nullptr);
    std::cin >> t;
    while (t--) solve();
}
```

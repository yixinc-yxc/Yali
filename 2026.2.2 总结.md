## B

最毒的一题。

想到了 `DP` ，但是想的是线性 `DP`  ~~因为这是J组，老师绝对不会第二题出多维DP~~

$dp_{i,j}$ 表示考虑前 $i$ 个糖果装在 $j$ 个盒子中的方案数。

而转移就是第 $i$ 个糖果有可能单独成盒，也有可能装在前面的盒子中。

第一种情况就是 $dp_{i - 1, j - 1}$ 的方案数。

而第二种就是 $(j - 1) * dp_{i - 1, j}$

所以 $dp_{i, j} = dp_{i - 1, j - 1} + (j - 1) * dp_{i - 1, j}$

```cpp
#include <bits/stdc++.h>

#define pb push_back
#define pf push_front

const ll INF = 1LL << 60LL;

const db eps = 1e-9;
std::mt19937_64 rng(std::chrono::system_clock::now().time_since_epoch().count());

const int N = 5010;
int n, k, dp[N][N];

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr), std::cout.tie(nullptr);
    std::cin >> n >> k;
    dp[0][0] = 1;
    rep(i, 1, n) rep(j, 1, std::min(i, k)) dp[i][j] = (dp[i - 1][j - 1] + 1ll * (i - 1) * dp[i - 1][j] % mod) % mod;
    std::cout << dp[n][k] << '\n';
}
```

## C

比较毒瘤。

看题面，吓哭了，以为是特难的图论，看数据范围，立马想到写 `floyd` ，然后 WA 了。

其实就是跑 $\min(n, k)$ 遍 `floyd`

关于 $\min(n, k)$ 的说明：在图上两点间经过了超过了 $n - 1$ 条边后一定不优，所以取 $\min$

```cpp
#include <bits/stdc++.h>

using ll = long long;
#define s second
#define f first
const int inf = 1e9;
const ll INF = 1LL << 60LL;
std::mt19937_64 rng(std::chrono::system_clock::now().time_since_epoch().count());

const int N = 80;
int n, m, k, q, f[N][N], dis[N][N], g[N][N];

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr), std::cout.tie(nullptr);
    std::cin >> n >> m;
    for (int i = 1; i <= n; i++) for (int j = 1; j <= n; j++) f[i][j] = dis[i][j] = g[i][j] = inf;
    for (int i = 1; i <= n; i++) f[i][i] = dis[i][i] = g[i][i] = 0;
    for (int i = 1; i <= m; i++) {
        int u, v, w;
        std::cin >> u >> v >> w;
        f[u][v] = std::min(f[u][v], w);
        dis[u][v] = std::min(dis[u][v], w), g[u][v] = std::min(g[u][v], w);
    }
    std::cin >> k >> q;
    k = std::min(k, n);
    for (int p = 1; p < k; p++) {
        for (int i = 1; i <= n; i++) for (int j = 1; j <= n; j++) f[i][j] = dis[i][j];
        for (int l = 1; l <= n; l++) for (int i = 1; i <= n; i++) for (int j = 1; j <= n; j++) f[i][j] = std::min(f[i][j], dis[i][l] + g[l][j]);
        for (int i = 1; i <= n; i++) for (int j = 1; j <= n; j++) dis[i][j] = f[i][j];
    }
    while (q--) {
        int u, v;
        std::cin >> u >> v;
        std::cout << (f[u][v] == inf ? -1 : f[u][v]) << '\n';
    }
}
```

```cpp
using ll = long long;
using ull = unsigned long long;
using i128 = __int128;
using db = double;
#define rep(i, a, b) for (int i = a; i <= b; i++)
#define frep(i, b, a) for (int i = b; i >= a; i--)
#define all(x) (x).begin(), (x).end()
#define pb push_back
#define pf push_front
#define mp make_pair
#define s second
#define f first
const int inf = 1e9;
const ll INF = 1LL << 60LL;
const int mod = 998244353;
const db eps = 1e-9;
std::mt19937_64 rng(std::chrono::system_clock::now().time_since_epoch().count());

const int N = 2e5 + 10;
int n, ans, a[N], v[N];
ll f[4 * N];
std::vector<int> b;

void add(int k, int l, int r, int pos, int val) {
    if (l == r) return f[k] += val, void();
    int mid = l + r >> 1;
    if (pos <= mid) add(k << 1, l, mid, pos, val);
    else add(k << 1 | 1, mid + 1, r, pos, val);
    f[k] = f[k << 1] + f[k << 1 | 1];
}

void find(int k, int l, int r, ll x) {
    if (l == r) return ans = l, void();
    int mid = l + r >> 1;
    if (f[k << 1] >= x) find(k << 1, l, mid, x);
    else find(k << 1 | 1, mid + 1, r, x - f[k << 1]);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr), std::cout.tie(nullptr);
    std::cin >> n;
    b.resize(n + 10);
    rep(i, 0, n - 1) std::cin >> b[i] >> a[i + 1], v[i + 1] = b[i];
    std::sort(all(b));
    b.erase(std::unique(all(b)), b.end());
    rep(i, 1, n) v[i] = std::lower_bound(all(b), v[i]) - b.begin();
    rep(i, 1, n) {
        add(1, 1, n, v[i], a[i]);
        ll x = f[1] + 1 >> 1;
        ans = -1;
        find(1, 1, n, x);
        std::cout << b[ans] << '\n';
    }
}
```
